==== GCE Xenial upgrade and fresh install test 17.1.46 -> 17.2.35 ====
# General test flow is as follows on xenial and artful
# part 1 Show current installed cloud-init with user-data and check performance
#   via tracebacks and cloud-init/systemd analyze
# part 2 upgrade cloud-init on a live system cloud-init init checking for
#   failures
# part 3 clean reboot and check performance (tracebacks and analyze)
# part 4 check specific bug fix behavior:

# Deployed via gcloud CLI with userdata
cat > sethostname.yaml << EOF
#cloud-config
ssh_import_id: [chad.smith]
hostname: SRU-worked
EOF

cat > setup_proposed.sh <<EOF
#/bin/bash
mirror=http://archive.ubuntu.com/ubuntu
echo deb \$mirror \$(lsb_release -sc)-proposed main | tee /etc/apt/sources.list.d/proposed.list
apt-get update -q;
apt-get install -qy cloud-init;
EOF


release=xenial
release_ver=1604
# Xenial instance with user-data
IMAGE_VERSION=`image-status gce $release | grep us-central | awk '{print $2}'`;

# part 1 deploy with user-data
gcloud compute instances create $release-sru-test --zone=us-central1-b --image daily-ubuntu-$release_ver-$release-v$IMAGE_VERSION --image-project ubuntu-os-cloud-devel --metadata-from-file user-data=sethostname.yaml;

GCE_VM=`gcloud compute instances list | grep $release-sru-test | awk '{print $5}'`
ssh ubuntu@$GCE_VM -- dpkg-query --show cloud-init;
ssh ubuntu@$GCE_VM sudo cat /run/cloud-init/result.json;
ssh ubuntu@$GCE_VM cloud-init status --long;
ssh ubuntu@$GCE_VM grep Trace /var/log/cloud-init.log;
ssh ubuntu@$GCE_VM systemd-analyze;
ssh ubuntu@$GCE_VM systemd-analyze blame;
ssh ubuntu@$GCE_VM cloud-init analyze show;
ssh ubuntu@$GCE_VM cloud-init analyze blame;
# part 2 upgrade cloud-init on live system
scp setup_proposed.sh ubuntu@$GCE_VM:.;
ssh ubuntu@$GCE_VM sudo bash ./setup_proposed.sh 2>&1 | egrep 'cloud-init';
ssh ubuntu@$GCE_VM sudo cloud-init init;
# part 3 clean boot cloud-init
ssh ubuntu@$GCE_VM sudo hostname SRU-didnt-work;
ssh ubuntu@$GCE_VM sudo cloud-init clean --logs --reboot;
ssh-keygen -f ".ssh/known_hosts" -R $GCE_VM;
ssh ubuntu@$GCE_VM grep Trace /var/log/cloud-init.log;
ssh ubuntu@$GCE_VM sudo cat /run/cloud-init/result.json;
ssh ubuntu@$GCE_VM cloud-init status --long;
ssh ubuntu@$GCE_VM sudo systemd-analyze blame;
ssh ubuntu@$GCE_VM cloud-init analyze show;
