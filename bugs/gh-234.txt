[Impact]
Some cloud environments pass multipart user-data as text/x-shellscript.
We need to make sure that it is accepted by cloud-init as user-data.

[Test Case]

```
#!/bin/sh
set -x
# Manually deploy on a lxc container

echo -e 'The userdata-mime file was generated by running the command:
tools/make-mime.py --attach userdata:x-shellscript  > userdata-mime

where userdata is defined as the following file:
#cloud-config
final_message: The system is finally up, after seconds $UPTIME seconds'

cat > userdata-mime << EOF
Content-Type: multipart/mixed; boundary="===============0816720877949849914=="
MIME-Version: 1.0

MIME-Version: 1.0

Content-Disposition: attachment; filename="userdata"

I2Nsb3VkLWNvbmZpZwpmaW5hbF9tZXNzYWdlOiBUaGUgc3lzdGVtIGlzIGZpbmFsbHkgdXAsIGFm
dGVyIHNlY29uZHMgJFVQVElNRSBzZWNvbmRzCg==

--===============0816720877949849914==--
EOF

cat > setup_proposed.sh <<EOF
#/bin/bash
mirror=http://archive.ubuntu.com/ubuntu
echo deb \$mirror \$(lsb_release -sc)-proposed main | tee /etc/apt/sources.list.d/proposed.list
apt-get update -q;
apt-get install -qy cloud-init;
EOF

recreate_issue() {
   series=$1

   name_recreate=test-$series-recreate
   lxc delete $name_recreate --force 2> /dev/null
   lxc launch ubuntu-daily:$series $name_recreate -c user.user-data="$(cat userdata-mime)"

   lxc exec $name_recreate -- cloud-init status --wait --long
   lxc exec $name_recreate -- cat /var/lib/cloud/seed/nocloud-net/user-data
   echo 'Verify that the mimetype used is not supported'
   if [ "$series" = "xenial" ]; then
       error_pattern="Exec format error"
   else
       error_pattern="Exec format error: b'/var/lib/cloud/instance/scripts/userdata'"
   fi

   lxc exec $name_recreate -- grep -A5 -B5 "$error_pattern" /var/log/cloud-init.log
   lxc exec $name_recreate -- grep 'The system is finally up, after seconds' /var/log/cloud-init-output.log
}

verify_issue_on_sru() {
   series=$1
   ref=$series-proposed
   name_proposed=test-$series-proposed

   ## upgrade cloud-init to proposed
   lxc delete $name_proposed --force 2> /dev/null
   lxc-proposed-snapshot --proposed --upgrade cloud-init --publish $series $name_proposed
   lxc init $ref $name_proposed -c user.user-data="$(cat userdata-mime)"
   lxc start $name_proposed
   lxc exec $name_proposed -- cloud-init status --wait --long
   lxc exec $name_proposed -- cat /var/lib/cloud/seed/nocloud-net/user-data
   echo 'Re test on current cloud-init to see if mimetype is now supported'
   lxc exec $name_proposed -- grep -A5 -B5 "$error_pattern" /var/log/cloud-init.log
   lxc exec $name_proposed -- grep 'The system is finally up, after seconds' /var/log/cloud-init-output.log
}

for SERIES in bionic eoan focal xenial; do
   echo '=== BEGIN ' $SERIES
   recreate_issue $SERIES
   verify_issue_on_sru $SERIES
done
```

[Regression Potential]
Maybe that send cloud-config info as text/x-shellscript would now be able to actually use them.
But otherwise, we believe there is no other regression associated with this.
